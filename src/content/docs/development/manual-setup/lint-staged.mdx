## Installation

<Tabs items={["Terminal", ".gitignore"]}>

<Tab value="Terminal">
<InstallTabs items={['npm', 'pnpm', 'yarn', 'bun']}>

```bash tab="npm" title="Terminal"
npm i -D lint-staged
echo "npm run lint-staged" > .husky/pre-commit
```

```bash tab="pnpm" title="Terminal"
pnpm i -D lint-staged
echo "pnpm lint-staged" > .husky/pre-commit
```

```bash tab="yarn" title="Terminal"
yarn add --dev lint-staged
echo "yarn run lint-staged" > .husky/pre-commit
```

```bash tab="bun" title="Terminal"
bun add --dev lint-staged
echo "burn run lint-staged" > .husky/pre-commit
```
